def count_substring(string, sub_string):
    C = 0
    x = -1
    while x < len(string):
        z = string.find(sub_string, x + 1)
        if z == -1:
            break
        C += 1
        x = z
    return C


if __name__ == '__main__':
    string = input().strip()
    sub_string = input().strip()
    
    count = count_substring(string, sub_string)
    print(count)
